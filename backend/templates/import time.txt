import time
import requests

def wait_for_server(url="http://localhost:5000/", timeout=30, interval=0.5):
    deadline = time.time() + timeout
    while time.time() < deadline:
        try:
            r = requests.get(url, timeout=2)
            if r.status_code < 500:
                return True
        except requests.RequestException:
            pass
        time.sleep(interval)
    raise RuntimeError(f"Server at {url} not available after {timeout}s")
